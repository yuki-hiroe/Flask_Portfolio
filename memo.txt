📁 最終的なプロジェクト構造
blog-app/
│
├── app.py                      # メインアプリケーション
├── blog.db                     # SQLiteデータベース（自動生成）
│
└── templates/
    ├── login.html              # 管理者ログイン
    ├── test_page.html          # テスト用ページ
    ├── admin/                  # 管理画面
    │   ├── base.html           # 管理画面ベース
    │   ├── dashboard.html      # ダッシュボード
    │   ├── posts.html          # 記事管理
    │   ├── post_form.html      # 記事作成・編集
    │   └── change_password.html # パスワード変更
    └── frontend/               # フロントエンド（読者向け）
        ├── base.html           # フロントエンドベース
        ├── home.html           # ホームページ
        ├── post_detail.html    # 記事詳細
        ├── posts_list.html     # 記事一覧
        ├── about.html          # プロフィール
        ├── contact.html        # お問い合わせ
        └── 404.html            # 404エラー
🚀 起動方法
bash# 1. アプリケーション起動
python app.py

# 2. ブラウザでアクセス
# フロントエンド: http://127.0.0.1:5000/
# 管理画面: http://127.0.0.1:5000/login
